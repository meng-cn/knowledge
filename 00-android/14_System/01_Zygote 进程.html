// frameworks/base/core/jni/AndroidRuntime.cpp

void AndroidRuntime::start(const char* className, const Vector<String8>& options, bool zygote)
{
    // 1. еҲӣе»ә JavaVM
    JavaVM* jvm;
    JNIEnv* env;
    if (startVm(&mJavaVM, &env, zygote) != 0) return;
    
    // 2. жіЁеҶҢ JNI ж–№жі•
    if (startReg(env) < 0) return;
    
    // 3. иҺ·еҸ– main ж–№жі•
    jclass stringClass = env->FindClass("java/lang/String");
    jclass startClass = env->FindClass(className);
    jmethodID startMeth = env->GetStaticMethodID(startClass, "main", 
        "([Ljava/lang/String;)V");
    
    // 4. и°ғз”Ё Java е…ҘеҸЈ
    env->CallStaticVoidMethod(startClass, startMeth, strArray);
}

        }
    }
}

Q7: Zygote зҡ„ Socket йҖҡдҝЎжңәеҲ¶жҳҜжҖҺж ·зҡ„пјҹ вҖӢ

еҸӮиҖғзӯ”жЎҲпјҡ Zygote дҪҝз”Ё Unix Domain Socket дёҺ AMS йҖҡдҝЎпјҡ

java
// ZygoteServer.java
public class ZygoteServer {
    private LocalServerSocket mServerSocket;
    
    public ZygoteServer() {
        // еҲӣе»ә ServerSocket
        mServerSocket = new LocalServerSocket("zygote");
    }
    
    public void runSelectLoop() {
        while (true) {
            // зӯүеҫ…иҝһжҺҘ
            LocalSocket peer = mServerSocket.accept();
            
            // еҲӣе»ә ZygoteConnection еӨ„зҗҶиҜ·жұӮ
            ZygoteConnection conn = new ZygoteConnection(peer);
            conn.processCommand(this, commands);
        }
    }
}
